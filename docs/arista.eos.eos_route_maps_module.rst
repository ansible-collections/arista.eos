    # veos#show running-config | section route-map

          config:
            - route_map: "mapmerge"
              entries:
                - description: "merged_map"
                  action: "permit"
                  sequence: 10
                  match:
                    router_id: 22
                - description: "newmap"
                  action: "deny"
                  sequence: 25
                  continue_sequence: 45
